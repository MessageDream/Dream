golang 获取当天起始时间

April 16 , 2013 at 22:55 am - - 78点击 - 1 评论 
package main

import (
	"fmt"
	"time"
)

func main() {

	t := time.Now()
	year, month, day := t.Date()
	today := time.Date(year, month, day, 0, 0, 0, 0, t.Location())
	fmt.Println(today)
}
